import ollama

def extract_skills_ollama(text):
    response = ollama.chat(model='mistral', messages=[
        {'role': 'user', 'content': f'Extract technical skills from this text as a Python list: {text}'}
    ])
    # Parse response to get the list
    return response['message']['content']
